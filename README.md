# Java_Switch
Syntax

switch(expression) {

  case x:
  
    // code block
    
    break;
    
  case y:
  
    // code block
    
    break;
    
  default:
  
    // code block
    
}

#Example

int day = 4;

switch (day)  {
  case 6:
  
    System.out.println("Today is Saturday");
    
    break;
    
  case 7:
  
    System.out.println("Today is Sunday");
    
    break;
    
  default:
  
    System.out.println("Looking forward to the Weekend");
}
